\section*{Solutions}
\begin{enumerate}
\item $
    (3-2i)+(-7+5i) = (3-7)+(-2+5)i \\
    = -4+3i
$
\item $
    (3-2i)-(-7+5i) = (3-(-7))-(-2-5)i \\
    = 10-7i
$
\item $
    (3-2i)(-7+5i) = (3)(-7)+(3)(5)i+(-2)(-7)i+(-2)(5)i^2 \\
    = -21+15i+14i-10i^2 \\
    = -10i^2+29i-21
$
\item \[
    \frac{1}{1-i} = \frac{1}{1-i}\left(\frac{1+i}{1+i}\right)
    = \frac{1+i}{1-i^2}
    = \frac{1+i}{2} \\
    = \frac{1}{2}+\frac{1}{2}(i)
\]
\item \[ \frac{3-2i}{-7+5i} \]
\item $ \overline{5-12i} $
\item $ |5-12i| $
\item $ |5+12i| $
\item $ i^{100} $
\item $ i^{49} $

Find all real or complex solutions:
\item $ 4x^2+9=0 $
\item $ x^2+z=-2 $

Illustrate on a graph the parallelogram or triangle law for the expressions:
\item $ (2+i)+(3-4i) $
\item $ (2+2i)-(-3+i) $
\item \text{Let $z=2+i$ and $w=3-4i$. Use a graph to illustrate $z,w,|z-w|$.}

\end{enumerate}
